<objective>
Create a GitHub Actions security scanning workflow that runs npm audit and npm outdated on every push to master and every pull request. This ensures continuous vulnerability detection and prevents insecure code from being deployed.
</objective>

<context>
@package.json
</context>

<tasks>

<task type="auto">
  <name>Task 1: Create GitHub security workflow</name>
  <files>.github/workflows/security.yml</files>
  <read_first>
    - package.json (to understand build scripts)
  </read_first>
  <action>
    Create .github/workflows/security.yml with the following content:
    ```yaml
    name: Security Scan

    on:
      push:
        branches: [master]
      pull_request:
        branches: [master]

    jobs:
      audit:
        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v4

          - name: Setup Node.js
            uses: actions/setup-node@v4
            with:
              node-version: '20'
              cache: 'npm'

          - name: Install dependencies
            run: npm ci

          - name: Run npm audit
            run: npm audit --audit-level=high
            continue-on-error: true

          - name: Check for outdated packages
            run: npm outdated --depth=0 || true

      build:
        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v4

          - name: Setup Node.js
            uses: actions/setup-node@v4
            with:
              node-version: '20'
              cache: 'npm'

          - name: Install dependencies
            run: npm ci

          - name: Build
            run: npm run build
    ```
    This workflow runs npm audit and build on every push and PR.
  </action>
  <acceptance_criteria>
    - .github/workflows/security.yml exists
    - Workflow runs on push to master and pull requests
    - Includes npm audit and npm run build steps
  </acceptance_criteria>
  <verify>
    <automated>ls -la .github/workflows/security.yml</automated>
  </verify>
  <done>GitHub security workflow created</done>
</task>
