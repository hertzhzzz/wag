    ```typescript
    import { test, expect } from '@playwright/test'

    test.describe('Navbar Sticky Behavior - Mobile', () => {
      const mobileViewport = { width: 320, height: 568 }

      test('navbar stays fixed during scroll on mobile', async ({ page }) => {
        await page.setViewportSize(mobileViewport)
        await page.goto('/')

        // Get the navbar element
        const navbar = page.locator('nav').first()

        // Verify navbar is visible before scroll
        await expect(navbar).toBeVisible()

        // Get initial navbar position
        const initialBox = await navbar.boundingBox()
        expect(initialBox).not.toBeNull()

        // Scroll down the page
        await page.evaluate(() => window.scrollTo(0, 500))

        // Wait for scroll to complete
        await page.waitForTimeout(300)

        // Verify navbar is still visible after scroll
        await expect(navbar).toBeVisible()

        // Verify navbar position is still at top (fixed)
        const afterScrollBox = await navbar.boundingBox()
        expect(afterScrollBox).not.toBeNull()

        // The navbar should still be at the top of viewport (y position close to 0)
        expect(afterScrollBox!.y).toBeLessThanOrEqual(5)

        // Verify hamburger menu is still clickable after scroll
        const hamburger = page.locator('nav button[aria-label*="Toggle"]').first()
        await expect(hamburger).toBeVisible()
        await expect(hamburger).toBeEnabled()
      })

      test('navbar stays fixed on Resources page during scroll', async ({ page }) => {
        await page.setViewportSize(mobileViewport)
        await page.goto('/resources')

        const navbar = page.locator('nav').first()
        await expect(navbar).toBeVisible()

        // Scroll down
        await page.evaluate(() => window.scrollTo(0, 500))
        await page.waitForTimeout(300)

        // Verify navbar still visible
        await expect(navbar).toBeVisible()

        const afterScrollBox = await navbar.boundingBox()
        expect(afterScrollBox!.y).toBeLessThanOrEqual(5)
      })
    })
    ```
